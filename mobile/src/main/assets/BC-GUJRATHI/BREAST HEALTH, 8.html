સ્તનનું કેન્સર
શસ્ત્રક્રિયા બાદ કસરતો

સ્તનના કેન્સરની શસ્ત્રક્રિયા પછીના દિવસો અને અઠવાડિયાઓ સુધી જે કરી શકાય તેવી કસરતોની રચના કરવામાં આવેલી છે. તેઓ મોટાભાગના લોકો માટે સાનુકુળ રહે છે. જો કોઈપણ જાતની ચાલુ રહેલી ચિંતાઓ હોય જેમ કે પહેલેથી રહેલ હાથ અથવા ખભા અંગેની ચિંતા, તો એ કસરતોની શરૂઆત કરતાં પહેલાં એક નિષ્ણાત સાથે પરામર્શ કરવાની જરૂર રહે છે.

પ્રસ્તાવના

કસરત શા માટે?

જે સ્ત્રીઓ અને ખાસ કરીને પુરુષોને સ્તનના કેન્સર માટે શસ્ત્ર ક્રિયા કરાવેલી હોય તેમણે શસ્ત્રક્રિયા અથવા રેડીયોથેરાપી પહેલાં તેમના હાથની જે રીતે હલન-ચલન થઈ શકતી હતી તે પાછી મેળવવામાં એ કસરતો દ્વારા મદદ મળી રહેશે. એ કસરતને લીધે કદાચ એ કોષપેશીને સ્વસ્થ બનવામાં પણ કદાચ મદદ મળી રહે, કારણ કે કસરત દરમિયાન જે ભાગ હલન-ચલન પામી રહેલો હોય તેને વધુ પ્રમાણમાં ઓક્સીજન પૂરો પાડવામાં આવે છે, અને તેથી દુ:ખાવો પણ ઓછો થાય છે. .

કસરતની શરૂઆત ક્યારે થવી જોઈએ?

શસ્ત્રક્રિયા થઈ ગયા બાદ શક્ય હોય તેટલી ઝડપથી કસરતની શરૂઆત થવી જ જોઈએ – આદર્શ રીતે તેના પછીના દિવસથી શરૂઆત થવી જોઈએ. એક વ્યક્તિ ધીરે ધીરે રોજ-બ-રોજની ક્રિયાઓ જેમ કે વેક્યુમ ક્લિનરનો ઉપયોગ કરવો, ક્પડાં ધોઈને ટાંગવા કે ડ્રાઇવિંગ કરવા જેવી ક્રિયાઓને ફરીથી શરૂ કરી શકે છે. એ કસરતો એક વ્યક્તિ જેટલી મર્યાદામાં આરામદાયક અનુભવ કરે તેટલી મર્યાદામાં જ થવી જોઈએ, અને શરૂઆતમાં ટૂંકા સમય માટે કરવી જોઈએ

એ કસરતોને કેટલા સમય માટે કરવાની જરૂર રહેશે?

જો સ્નાયુઓનો ઉપયોગ ના કરવામાં આવે તો સ્નાયુઓ ઝકડાઇ જાય છે. હાથની સખતાઇ અથવા ઝકડન એ શસ્ત્રક્રિયા અથવા રેડીયોથેરાપી થયાના લાંબા સમયગાળા બાદ થઈ શકે છે તેથી આ બધી કસરતોને રોજ-બ-રોજની ક્રિયાઓના ભાગરૂપે કરતાં રહેવું એ મહત્વનું છે.

પ્રાથમિક કસરતો

શરૂઆતમાં દરેક કસરતને ત્રણ થી પાંચ વખત કરવી જોઈએ અને ધીમે ધીમે, એ વ્યક્તિને જ્યારે યોગ્ય અનુભવ થાય, ત્યારે 10 વખત સુધી વધારી શકાય છે. આદર્શ રીતે, આ કસરતોની શરૂઆત શસ્ત્રક્રિયા પછીના દિવસથી કરવામાં આવે છે

ખભાને ગોળાકાર રીતે હલાવવા

આ કસરતનો હૂંફ મેળવવા માટે ઉપયોગ કરો. નીચે બેસીને, તમારા હાથના પંજાને તમારા ખોળામાં રાખો, તમારા ખભાને તમારા કાન તરફ ઊંચા કરો, પછી તેને પાછળની તરફ અને નીચેની તરફ ગોળાકાર રીતે ફેરવો. આટલી જ સંખ્યામાં એ રીતે જ આગળની તરફ અને પાછળની તરફ કસરત કરો.



વાળ ઓળવા

તમે તમારા વાળ ઓળી રહ્યાં છો અથવા તેમાં દાંતિયો ફેરવી રહ્યાં છો તેવી કલ્પના કરો. તમારા માથાને સીધું રાખો અને તમારા હાથને ઊંચો કરો, ત્યારે તમારા હાથની કોણીને તમારાથી દૂર રાખવી જોઈએ. તમે તમારા માથાના વાળને આગળ થી પાછળની તરફ ઓળી રહ્યાં છો અથવા તેમાં દાંતિયો ફેરવી રહ્યાં છો એ રીતે અભિનય કરો. માત્ર એક જ બાજુની સાથે શરૂઆત કરો, પછી ધીમે ધીમે સમગ્ર માથા સુધી વધારતા જાઓ. વધુ પડતું કરશો નહીં, પરંતુ પ્રયત્ન કરવાનું ચાલુ રાખશો.



સહાયની સાથે ઊંચકવું

ઊભા રહીને, તમારા હાથની ચોકડી વાળો, તમારા અસર પામેલા હાથની કોણીની નીચે અસર નહીં પામેલા હાથના પંજાને રાખો, બન્ને કોણીઓ વળેલી હોવી જોઈએ. તમારા અસર પામેલા હાથને આ રીતે ટેકો મળવાની સાથે, તમારા હાથને એ સ્થિતિમાં ખભાના સ્તર સુધી ઊંચકાવો – જો તમે કરી શકો તો તેનાથી પણ વધુ ઉંચકો. તમારા હાથને એ સ્થિતિમાં ધીમે ધીમે નીચે લાવો.



પીઠનું ખેંચાણ

તમારા હાથને બહારની બાજુએ રાખો અને તેને કોણીએથી વાળો. પછી ધીમે ધીમે તમારી પીઠ પર ખભાના હાડકાના નીચેના ભાગ સુધી પહોંચો



હાથને વાળો

ઊભાં રહીને અથવા બેસીને, તમારા બન્ને હાથને ઉપરની તરફ એવી રીતે ઊંચા કરો જેથી તેનો તમારા શરીર સાથે કાટખૂણો બને. તમારા હાથની કોણીઓ વાળો અને હળવેથી તમારા હાથના પંજાને ખભા પર રાખો. તમારા હાથને ધીમે ધીમે નીચેની તરફ લાવો, ફરીથી તેને ઉપરની તરફ લાવો. .



આગળ પડતી કસરતો

આ કસરતો એ પ્રવાહી નિકળી ગયાં અથવા બહાર કાઢી લેવાયાના થોડાં અઠવાડિયા બાદ જ કરવી જોઈએ.

હાથને ઊંચકવા

તમારા માથાને ટેકો મળી રહે એ રીતે તકિયો રાખીને એક પથારી પર થવા ભોંતળીયા પર સુઈ જાઓ. તમારા હાથના પંજાઓને એકબીજા સાથે જકડીલો, તમારી કોણીઓને સીધી રાખો, તમારા હાથને ઊંચકો અને તેને તમારા માથાની ઉપર, જ્યાં સુધી તમે આરામદાયક અનુભવો ત્યાં સુધી ઊંચકો. તે સ્થિતિમાં દશની ગણતરી કરો ત્યાં સુધી જાળવી રાખો, પછી તમારા હાથને ધીમે-ધીમે નીચે લાવો. સુતા હોવ ત્યારે, ખરેખર ઊંડા, ત્રણ કે ચાર શ્વાસ લો અને તમારા ખભાને આરામ પહોંચાડવા પર ધ્યાન કેન્દ્રિત કરો જેથી તેઓ તમારા કાન તરફ ખેંચાયેલ ના રહે.



દિવાલ પર ચડાઇ

સમતુલા જાળવવા માટે તમારા પગ પહોળા રાખીને, એક દિવાલ તરફ મોઢું રાખીને તેની નજીક ઊભા રહો. તમારા હાથના પંજાની સાથે ખભાના સ્તરથી શરૂઆત કરો, ત્યાર બાદ તમને દબાણ પડે નહીં એ રીતે ધીમે ધીમે એ હાથને તમે જેટલા કરી શકો તેટલા એ જ સ્થિતિમાં ઊંચા કરો. દશની ગણતરી થાય ત્યાં સુધી એ હાથને ત્યાં જાળવી રાખો, પછી એ કસરતનું પુનરાવર્તન કરતાં પહેલાં તમારા હાથની ખભાના સ્તર સુધી પાછાં સરકાવો. તમે દિવાલ પર જે સ્થાન સુધી પહોંચ્યા હોવ તેની નોંધ કરો અથવા તેને અંકિત કરો અને દરેક વખતે થોડા ઊંચા સ્થાન પર જવાનો પ્રયત્ન કરો.



પીઠને લૂછવી

એક ટુવાલ અથવા તેવા જ કોઈ કપડાંની સાથે, એક સૌમ્ય રીતે પીઠ-લૂછવાની ગતીનો ઉપયોગ કરો. એ સ્થિતિને ઉલ્ટાવો જેથી તમારો અન્ય હાથ પ્રમાણમાં ઊંચી કક્ષા પર પહોંચે.



કોણીને દબાણ

તમારા માથાની પાછળ તમારા હાથના પંજા રહે અને તમારી કોણીઓ બહારની બાજુએ રહે એ રીતે તમારી પીઠભર સુઈ જાઓ. સૌમ્ય રીતે તમારી કોણીઓને જ્યાંસુધી આરામદાયક અનુભવ રહે ત્યાં સુધી નીચેની તરફ ધકેલો. દશની ગણતરી થાય ત્યાં સુધી એ સ્થિતિને જાળવી રાખો, પછી આરામ કરો. જો તમે રેડીયોથેરાપી લેવા માટે જાઓ તો આ કસરત ખરેખર ઉપયોગી બની રહે છે કારણ કે એ સારવારમાં ઘણી વખત તમારે એ સ્થિતિમાં રહેવાની જરૂર પડે છે.
